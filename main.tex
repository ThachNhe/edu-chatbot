	\begin{abstract}
		\pagenumbering{arabic}
		\pagestyle{fancy}
		
		Trong bối cảnh chuyển đổi số trong giáo dục, nhu cầu sử dụng các công cụ hỗ trợ dạy học thông minh ngày càng trở nên rõ rệt, đặc biệt với những môn học có tính thực hành và đòi hỏi giải thích thường xuyên như Tin học 12. Từ thực tế đó, đề tài ``Xây dựng chatbot hỗ trợ giáo viên trong giảng dạy Tin học 12'' được thực hiện nhằm xây dựng một hệ thống web hỗ trợ giáo viên trong các nghiệp vụ đang có nhu cầu cao như hỏi đáp với AI, tạo đề thi trắc nghiệm bằng AI, tổ chức phòng thi trực tuyến và theo dõi kết quả học sinh.
		
		Hệ thống được xây dựng theo mô hình client -- server. Phần frontend sử dụng React, TypeScript và Vite để xây dựng giao diện; phần backend sử dụng FastAPI để cung cấp API và WebSocket; dữ liệu được lưu trữ trên PostgreSQL thông qua SQLAlchemy và Alembic. Chức năng chatbot sử dụng mô hình ngôn ngữ lớn kết hợp FAISS để truy xuất ngữ cảnh từ bộ dữ liệu học liệu đã được lập chỉ mục. Đề thi được sinh bằng AI dưới dạng câu hỏi trắc nghiệm, cho phép giáo viên lưu lại, chỉnh sửa, tạo phòng thi và công bố cho học sinh làm bài trực tuyến.
		
		Về mặt chức năng, hệ thống hiện hỗ trợ ba nhóm người dùng chính. Giáo viên có thể đăng nhập, khôi phục mật khẩu, hội thoại với AI, xem lịch sử hội thoại, tạo đề thi, tạo phòng thi, theo dõi dashboard và xem thống kê kết quả học sinh. Học sinh có thể truy cập phòng thi qua liên kết, xác thực mã học sinh, làm bài và nhận điểm sau khi nộp. Quản trị viên có thể xem số liệu tổng quan, thêm giáo viên, thêm học sinh, khóa hoặc mở khóa tài khoản khi cần thiết.
		
		Kết quả đạt được cho thấy hệ thống đã đáp ứng tốt các luồng chính của bài toán đặt ra và có khả năng triển khai tiếp trong môi trường thực tế. Bên cạnh đó, đồ án cũng chỉ ra một số nội dung cần phát triển thêm như hoàn thiện quản lý bài học, tích hợp gửi email khôi phục mật khẩu trong môi trường thực, mở rộng nguồn tri thức và tối ưu giao diện trên thiết bị di động.
	\end{abstract}

	\section{KHẢO SÁT VÀ PHÂN TÍCH YÊU CẦU}
	
	\subsection{Mục đích khảo sát}
	Mục đích khảo sát của đề tài là làm rõ nhu cầu thực tế của bài toán và xác định phạm vi triển khai phù hợp với nguồn lực hiện có. Nội dung khảo sát tập trung vào các vấn đề sau:
	
	\begin{itemize}
		\item Xác định những khó khăn mà giáo viên Tin học 12 thường gặp trong quá trình chuẩn bị bài, trả lời câu hỏi và tổ chức kiểm tra đánh giá.
		\item Xác định mức độ cần thiết của một công cụ AI có thể hỗ trợ hỏi đáp, gợi ý nội dung và tạo đề thi trắc nghiệm.
		\item Xác định các chức năng cốt lõi cần ưu tiên triển khai trong phiên bản đầu tiên của hệ thống.
		\item Đánh giá cách tổ chức giao diện sao cho phù hợp với ba nhóm người dùng: giáo viên, học sinh và quản trị viên.
	\end{itemize}
	
	\subsection{Khảo sát}
	
	\subsubsection{Đối tượng khảo sát}
	Khảo sát của đề tài hướng đến ba nhóm đối tượng chính:
	
	\begin{itemize}
		\item \textbf{Giáo viên Tin học THPT}: là nhóm sử dụng trực tiếp các chức năng hỏi đáp AI, tạo đề thi, tạo phòng thi và theo dõi kết quả học sinh.
		\item \textbf{Học sinh lớp 12}: là nhóm tham gia làm bài trong các phòng thi do giáo viên tạo ra và nhận phản hồi kết quả sau khi nộp bài.
		\item \textbf{Quản trị viên}: là nhóm quản lý tài khoản giáo viên, học sinh và theo dõi số liệu tổng quan của hệ thống.
	\end{itemize}
	
	\subsubsection{Khảo sát nghiệp vụ}
	Qua quá trình phân tích yêu cầu và đối chiếu với các luồng nghiệp vụ thực tế, nhóm xác định các nhu cầu nổi bật như sau:
	
	\paragraph{Đối với giáo viên}
	\begin{itemize}
		\item Cần một công cụ có thể trả lời nhanh các câu hỏi về nội dung Tin học THPT, giúp tiết kiệm thời gian tra cứu tài liệu.
		\item Cần rút ngắn thời gian biên soạn câu hỏi trắc nghiệm, đồng thời vẫn bảo đảm có thể chỉnh sửa trước khi sử dụng.
		\item Cần một cách đơn giản để tổ chức làm bài trực tuyến và theo dõi điểm số học sinh theo từng đề thi đã giao.
	\end{itemize}
	
	\paragraph{Đối với học sinh}
	\begin{itemize}
		\item Cần một cách vào phòng thi trực quan, không yêu cầu thao tác kỹ thuật phức tạp.
		\item Cần giao diện làm bài rõ ràng, có bộ đếm thời gian và thông báo kết quả sau khi nộp bài.
	\end{itemize}
	
	\paragraph{Đối với quản trị viên}
	\begin{itemize}
		\item Cần giao diện riêng để tạo tài khoản giáo viên, thêm học sinh và khóa hoặc mở khóa người dùng.
		\item Cần xem được số liệu tổng quan như tổng số học sinh, giáo viên và đề thi đã được tạo trong hệ thống.
	\end{itemize}

	\subsubsection{Phân tích yêu cầu}
	Từ khảo sát và đối chiếu với phạm vi triển khai thực tế của codebase, nhóm xác định hệ thống cần đáp ứng những yêu cầu sau:
	
	\paragraph{Yêu cầu đối với giao diện}
	\begin{itemize}
		\item Giao diện phải rõ ràng, dễ sử dụng và tách biệt theo nhóm vai trò.
		\item Các thao tác chính như đăng nhập, hỏi AI, tạo đề, tạo phòng thi và xem thống kê phải được truy cập nhanh.
		\item Màn hình làm bài dành cho học sinh phải tối giản, tập trung vào câu hỏi và thời gian làm bài.
	\end{itemize}
	
	\paragraph{Chức năng của giáo viên}
	\begin{itemize}
		\item Đăng nhập và đăng xuất hệ thống bằng email và mật khẩu.
		\item Gửi yêu cầu khôi phục mật khẩu và đặt lại mật khẩu bằng token.
		\item Hội thoại với AI, lưu lịch sử hội thoại và mở lại các cuộc hội thoại cũ.
		\item Tạo đề thi trắc nghiệm bằng AI, chỉnh sửa câu hỏi, lưu nháp hoặc công bố đề thi.
		\item Tạo phòng thi, nhận liên kết truy cập và bật hoặc tắt trạng thái phòng thi.
		\item Xem dashboard tổng quan và thống kê kết quả học sinh.
	\end{itemize}
	
	\paragraph{Chức năng của học sinh}
	\begin{itemize}
		\item Truy cập phòng thi bằng liên kết chứa mã phòng.
		\item Nhập thông tin xác thực và làm bài trực tuyến.
		\item Nộp bài và nhận điểm ngay sau khi hệ thống chấm tự động.
	\end{itemize}
	
	\paragraph{Chức năng của quản trị viên}
	\begin{itemize}
		\item Xem số liệu tổng quan về giáo viên, học sinh và đề thi.
		\item Thêm học sinh mới, thêm giáo viên mới.
		\item Khóa hoặc mở khóa tài khoản giáo viên và học sinh.
	\end{itemize}
	
	\paragraph{Phạm vi chưa hoàn thiện hoàn toàn trong phiên bản hiện tại}
	\begin{itemize}
		\item Phần bài học hiện mới ở mức giao diện danh mục và mô hình dữ liệu, chưa có luồng CRUD đầy đủ trên API như các module chính.
		\item Chức năng quên mật khẩu đã có API sinh token và đường dẫn đặt lại mật khẩu, nhưng trong môi trường phát triển đường dẫn này hiện được ghi ra log thay vì gửi email thật.
	\end{itemize}
	
	\section{PHÂN TÍCH VÀ THIẾT KẾ HỆ THỐNG}
	
	\subsection{Xác định bài toán}
	\paragraph{Tên đề tài}
	Xây dựng chatbot hỗ trợ giáo viên trong giảng dạy Tin học 12.
	
	\paragraph{Mô tả bài toán}
	Đề tài hướng tới việc xây dựng một hệ thống web hỗ trợ giáo viên Tin học 12 trong các hoạt động trọng tâm gồm: hỏi đáp với AI, tạo đề kiểm tra trắc nghiệm bằng AI, tạo phòng thi trực tuyến cho học sinh và theo dõi kết quả học tập. Hệ thống cần hoạt động trên nền web, có phân quyền người dùng rõ ràng và bảo đảm dữ liệu được lưu trữ tập trung.
	
	\paragraph{Kết quả dự kiến đạt được}
	\begin{itemize}
		\item Xây dựng được bản phân tích thiết kế hệ thống và mô hình cơ sở dữ liệu.
		\item Xây dựng được website gồm các vai trò giáo viên, học sinh và quản trị viên.
		\item Hoàn thiện được các luồng nghiệp vụ chính: chat với AI, tạo đề thi, tạo phòng thi, làm bài thi trực tuyến và xem thống kê.
	\end{itemize}
	
	\paragraph{Đối với giáo viên}
	\begin{itemize}
		\item Đăng nhập hệ thống.
		\item Thực hiện quên mật khẩu và đặt lại mật khẩu.
		\item Hỏi đáp với AI và xem lại lịch sử hội thoại.
		\item Tạo đề thi bằng AI, chỉnh sửa và lưu đề thi.
		\item Tạo phòng thi cho học sinh và xem thống kê kết quả.
		\item Xem dashboard tổng quan.
	\end{itemize}
	
	\paragraph{Đối với học sinh}
	\begin{itemize}
		\item Tham gia làm bài thi trực tuyến qua phòng thi được giáo viên chia sẻ.
		\item Nhận kết quả sau khi nộp bài.
	\end{itemize}
	
	\paragraph{Đối với quản trị viên}
	\begin{itemize}
		\item Quản lý giáo viên và học sinh.
		\item Theo dõi số lượng đối tượng và số lượng đề thi của toàn hệ thống.
	\end{itemize}

	\section{XÂY DỰNG CHƯƠNG TRÌNH}
	
	\subsection{Công nghệ sử dụng}
	
	\subsubsection{Python \& FastAPI}
	FastAPI \cite{fastapi} là framework web chính được sử dụng ở backend để xây dựng API và WebSocket. Việc sử dụng FastAPI giúp hệ thống dễ tổ chức theo module, hỗ trợ xác thực, xử lý dữ liệu đầu vào và phát triển nhanh các endpoint cho từng tính năng.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item Cung cấp API cho đăng nhập, quên mật khẩu, tạo đề thi, tạo phòng thi, thống kê và quản trị.
		\item Cung cấp WebSocket cho luồng chat thời gian thực với AI.
		\item Kết hợp tốt với Pydantic để kiểm tra dữ liệu đầu vào và đầu ra.
	\end{itemize}
	
	\subsubsection{PostgreSQL}
	PostgreSQL \cite{postgresql} được sử dụng làm hệ quản trị cơ sở dữ liệu quan hệ cho toàn bộ hệ thống. Đây là nơi lưu trữ thông tin người dùng, đề thi, câu hỏi, phòng thi, hội thoại và điểm số.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item Lưu trữ dữ liệu nghiệp vụ của giáo viên, học sinh và quản trị viên.
		\item Đảm bảo truy vấn ổn định cho các chức năng dashboard, thống kê và lịch sử hoạt động.
		\item Dễ dàng mở rộng khi số lượng đề thi, câu hỏi và kết quả học tập tăng lên.
	\end{itemize}
	
	\begin{figure}[H]

		\caption{PostgreSQL}
		\label{fig:postgres}

	\subsubsection{SQLAlchemy \& Alembic}
	SQLAlchemy được sử dụng để ánh xạ các đối tượng Python với bảng dữ liệu trong PostgreSQL. Alembic được sử dụng để quản lý migration, giúp đồng bộ cấu trúc dữ liệu trong quá trình phát triển.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item Định nghĩa model cho các bảng như users, students, exams, scores, conversations.
		\item Giúp viết truy vấn rõ ràng, dễ bảo trì và gắn với logic nghiệp vụ ở backend.
		\item Hỗ trợ thay đổi cấu trúc cơ sở dữ liệu theo từng phiên bản của đề tài.
	\end{itemize}
	
	\subsubsection{React, TypeScript \& Vite}
	Frontend của hệ thống được xây dựng bằng React, TypeScript và Vite \cite{vite}. React hỗ trợ xây dựng giao diện theo component, TypeScript giúp kiểm soát kiểu dữ liệu, còn Vite cung cấp môi trường phát triển nhanh và tối ưu quá trình build.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item Xây dựng giao diện cho giáo viên, học sinh và quản trị viên.
		\item Tổ chức các màn hình theo route rõ ràng như login, dashboard, chat, exam, stats, admin.
		\item Bảo đảm mã nguồn frontend rõ ràng, dễ mở rộng và giảm lỗi nhờ kiểm tra kiểu dữ liệu.
	\end{itemize}

	\subsubsection{TanStack Router, React Query và Zustand}
	Ngoài bộ công nghệ chính, frontend còn sử dụng TanStack Router để định tuyến, React Query để quản lý dữ liệu bất đồng bộ và Zustand để quản lý trạng thái cục bộ. Đây là những thành phần giúp kiến trúc frontend gọn, rõ ràng và dễ bảo trì.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item TanStack Router quản lý các route như \texttt{/dashboard}, \texttt{/chat}, \texttt{/exam}, \texttt{/stats}, \texttt{/admin/students}.
		\item React Query quản lý các request như lấy danh sách đề thi, thống kê, người dùng quản trị.
		\item Zustand lưu các trạng thái dùng chung như giao diện, thông báo và một số cài đặt trong ứng dụng.
	\end{itemize}
	
	\subsubsection{OpenAI, FAISS và WebSocket}
	Đây là nhóm công nghệ quan trọng nhất cho chức năng AI của hệ thống. Chatbot sử dụng mô hình ngôn ngữ lớn thông qua OpenAI, FAISS để truy xuất ngữ cảnh từ bộ dữ liệu đã lập chỉ mục, và WebSocket để duy trì trao đổi thời gian thực giữa frontend và backend.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item FAISS lưu trữ vector của bộ học liệu, cho phép tìm đoạn nội dung liên quan tới câu hỏi của người dùng.
		\item OpenAI được sử dụng cho hai luồng: sinh câu trả lời chat và sinh câu hỏi trắc nghiệm cho đề thi.
		\item WebSocket giúp màn hình chat phản hồi theo thời gian thực và hỗ trợ nạp lại lịch sử hội thoại cũ.
	\end{itemize}
	
	\subsubsection{Docker Compose, MailHog và SMTP}
	Hệ thống được cấu hình để có thể chạy bằng Docker Compose với các dịch vụ PostgreSQL, backend, frontend và MailHog. MailHog được dùng làm hộp thư giả lập trong quá trình phát triển để kiểm tra email tài khoản giáo viên.
	
	\paragraph{Vai trò trong hệ thống}
	\begin{itemize}
		\item Tạo môi trường chạy thống nhất và thuận tiện cho phát triển.
		\item Hỗ trợ kiểm thử chức năng gửi email khi quản trị viên tạo tài khoản giáo viên.
		\item Dễ dàng mở rộng sang môi trường triển khai thật bằng cách thay thế cấu hình SMTP.
	\end{itemize}

	\subsection{Kiến trúc triển khai chương trình}
	Kiến trúc hệ thống gồm các thành phần chính sau:
	
	\begin{itemize}
		\item \textbf{Frontend}: ứng dụng React hiển thị giao diện cho giáo viên, học sinh và quản trị viên.
		\item \textbf{Backend}: FastAPI cung cấp API và WebSocket, xử lý xác thực, quản lý đề thi, chấm điểm và quản trị.
		\item \textbf{Database}: PostgreSQL lưu trữ dữ liệu nghiệp vụ.
		\item \textbf{AI layer}: OpenAI và FAISS xử lý hỏi đáp và sinh đề thi.
		\item \textbf{Mail service}: MailHog hoặc SMTP phục vụ gửi email tài khoản giáo viên.
	\end{itemize}
	
	Trong luồng chat, backend nhận câu hỏi từ WebSocket, truy xuất ngữ cảnh liên quan từ chỉ mục FAISS và gửi prompt tới mô hình AI. Trong luồng tạo đề thi, backend gửi prompt sinh câu hỏi trắc nghiệm cho mô hình AI, sau đó frontend cho phép giáo viên chỉnh sửa trước khi lưu. Trong luồng làm bài thi, học sinh truy cập phòng thi công khai, backend kiểm tra mã học sinh, chấm tự động và lưu điểm vào bảng \texttt{scores}.

	\section{KẾT LUẬN VÀ KIẾN NGHỊ}
	
	Sau quá trình nghiên cứu, thiết kế và phát triển, đồ án đã xây dựng được một hệ thống web hỗ trợ giáo viên trong giảng dạy Tin học 12 với các chức năng trọng tâm đã hoạt động theo đúng codebase hiện có. Cụ thể, hệ thống đã triển khai được luồng đăng nhập và xác thực người dùng, khôi phục mật khẩu bằng token, hỏi đáp với AI qua WebSocket, lưu lịch sử hội thoại, tạo đề thi bằng AI, tạo phòng thi trực tuyến, tổ chức làm bài cho học sinh, chấm điểm tự động, hiển thị dashboard, thống kê học tập và quản trị tài khoản giáo viên, học sinh. Đây là những chức năng có giá trị sử dụng thực tế, hỗ trợ giáo viên tiết kiệm thời gian trong khâu chuẩn bị và đánh giá.
	
	Bên cạnh những kết quả đạt được, hệ thống vẫn còn một số hạn chế cần được nhìn nhận đúng với trạng thái triển khai hiện tại:
	
	\begin{itemize}
		\item Chất lượng phản hồi của chatbot phụ thuộc vào dữ liệu đã được lập chỉ mục trong bộ học liệu và khả năng của mô hình AI.
		\item Chức năng quên mật khẩu đã có luồng tạo token và đặt lại mật khẩu, nhưng việc gửi email khôi phục trong môi trường phát triển chưa được hoàn thiện như luồng gửi email tài khoản giáo viên.
		\item Phần bài học hiện chủ yếu ở mức danh mục và mô hình dữ liệu, chưa có API CRUD đầy đủ như các module chat, đề thi và thống kê.
		\item Giao diện học sinh hiện thu thập thêm lớp và email, nhưng backend hiện mới sử dụng mã học sinh để xác thực và lưu kết quả thi.
		\item Giao diện trên thiết bị di động vẫn cần tiếp tục tối ưu để sử dụng thuận tiện hơn trên nhiều kích thước màn hình.
	\end{itemize}
	
	Từ những hạn chế trên, nhóm đề xuất một số hướng phát triển tiếp theo như sau:
	
	\begin{itemize}
		\item Hoàn thiện module bài học và quản lý tri thức, cho phép cập nhật học liệu trực tiếp từ giao diện quản trị hoặc giáo viên.
		\item Tích hợp luồng gửi email thực tế cho chức năng quên mật khẩu và các thông báo hệ thống khác.
		\item Mở rộng nguồn dữ liệu học liệu, bổ sung nhiều bộ sách và tài liệu ôn tập hơn để nâng cao chất lượng phản hồi của chatbot.
		\item Nâng cấp phần thống kê, bổ sung lọc dữ liệu sâu hơn theo lớp, chủ đề hoặc thời gian kiểm tra.
		\item Tối ưu giao diện và trải nghiệm người dùng trên thiết bị di động.
	\end{itemize}
	
	Nhìn chung, đồ án đã đạt được mục tiêu chính là xây dựng một nền tảng hỗ trợ giảng dạy có ứng dụng thực tế, đồng thời tạo tiền đề kỹ thuật tốt để tiếp tục phát triển thành một hệ thống đầy đủ hơn trong tương lai.
